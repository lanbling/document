# 防抖函数



## 定义

一段时间内频繁触发的事件，但是不想频繁执行回调函数。

等用户停止交互一段时间后，再来执行回调函数。

关键函数：setTimeout



## 频繁触发的事件

- input框输入
- window的resize事件，scroll事件
- 点击事件



## 基础写法

```js
function debounce(fn, deley){
    var timer = null;
    return function(){
        if(timer){
            clearTimeout(timer);
            timer = null;
        }
        timer = setTimeout(function(){
            timer = null;
            fn();
        }, deley);
    }
}
```

缺点：第一次没法执行，也没有考虑参数情况



## 改进版本：优化第一次调用

```js
function debounce(fn, deley, leading){
    var timer = null;
    return function(){
        //清空定时器是第一步
        if(timer){
            clearTimeout(timer);
            timer = null;
        }
        var _this = this;
        var _arguments = arguments;
        
        //判断第一次执行，使用定时器来进行判断
        if(leading){
            var invoked = false;  //判断是否被调用
            //第一次进入，timer为null，执行函数，将invoked设为true
            if(!timer){
                fn.apply(_this, _arguments);
                invoked = true;
            }
            timer = setTimeout(function(){
                timer = null;   //执行完最后一次后，将timer初始化
                fn.apply(_this, _arguments);
            }, deley);
        }else{
            timer = setTimeout(function(){
                timer = null;
                fn.apply(_this, _arguments);
            }, deley);
        }
    }
}
```

缺点：没有考虑取消定时器的情况



## 改进版本2：添加取消定时器的功能

```js
function debounce(fn, deley, leading){
    var timer = null;
    
    function handleFn(){
        //清空定时器是第一步
        if(timer){
            clearTimeout(timer);
            timer = null;
        }
        var _this = this;
        var _arguments = arguments;
        
        //判断第一次执行，使用定时器来进行判断
        if(leading){
            var invoked = false;  //判断是否被调用
            //第一次进入，timer为null，执行函数，将invoked设为true
            if(!timer){
                fn.apply(_this, _arguments);
                invoked = true;
            }
            timer = setTimeout(function(){
                timer = null;   //执行完最后一次后，将timer初始化
                fn.apply(_this, _arguments);
            }, deley);
        }else{
            timer = setTimeout(function(){
                timer = null;
                fn.apply(_this, _arguments);
            }, deley);
        }
    }
    
    //函数也可以添加属性，惊不惊喜，意不意外。
    handleFn.cancel = function(){
        if(timer){
            clearTimeout(timer);
        }
    }
    
    return handleFn;
}
```

## 改进版本3：如果函数有返回值

```js
function debounce(fn, deley, options) {
    var timer = null;
    var leading = options.leading || false;
    var resultFn = options.resultFn;

    function handleFn() {
        //清空定时器是第一步,这里不要把timer设为null，不然会一直走第一次的步骤。
        //只有最后一次被执行了，才需要初始化
        if (timer) {
            clearTimeout(timer);
        }
        var _this = this;
        var _arguments = arguments;

        //判断第一次执行，使用定时器来进行判断
        if (leading) {
            var invoked = false;  //判断是否被调用
            //第一次进入，timer为null，执行函数，将invoked设为true
            if (!timer) {
                callbackFn(_this, _arguments);
                invoked = true;
            }
            timer = setTimeout(function () {
                timer = null;   //执行完最后一次后，将timer初始化
                callbackFn(_this, _arguments);
            }, deley);
        } else {
            timer = setTimeout(function () {
                timer = null;
                callbackFn(_this, _arguments);
            }, deley);
        }
    }

    //函数也可以添加属性，惊不惊喜，意不意外。
    handleFn.cancelFn = function () {
        if (timer) {
            clearTimeout(timer);
        }
    }

    function callbackFn(content, arguments) {
        var res = fn.apply(content, arguments);
        if (resultFn) {
            resultFn(res);
        }
    }
    return handleFn;
}
```
