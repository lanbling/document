# 节流

# 定义

一段时间内频繁触发事件，但是不是频繁执行函数。

设置一个时间间距，每隔一个间距执行一次，便是节流。

关键函数：new Date().getTime 、 setTimeout

# 基础写法

function throttle(fn, duration){
    var last = 0;
    return function(){
        var now = new Date().getTime();
        if(now - last >= duration){
            fn();
            last = now;
        }
    }
}

缺点：最后的一次函数没执行，没有考虑函数参数

# 改进版本1

function throttle(fn, duration, options) {
    var last = 0;
    var timer = null;
    var trailing = options.trailing || false; //最后一次是否调用

    function handleFn() {
        var _this = this;
        var _arguments = arguments;
        var now = Date.now();
        if (now - last >= duration) {
            if (timer) {
                timer = null;
                clearTimeout(timer);
            }
            fn.apply(_this, _arguments);
            last = now;
        } else if (timer === null && trailing) {
            timer = setTimeout(function () {
                timer = null;
                fn.apply(_this, _arguments);
            }, duration);
        }
    }

    return handleFn;
}

缺点：还需要加上取消定时器，以及考虑函数有返回值的情况

# 改进版本2

 function throttle(fn, duration, options) {
     var last = 0;
     var timer = null;
     var trailing = options.trailing || false; //最后一次是否调用
     var resultFn = options.resultFn;

     function handleFn() {
         var _this = this;
         var _arguments = arguments;
         var now = Date.now();
         if (now - last >= duration) {
             if (timer) {
                 timer = null;
                 clearTimeout(timer);
             }
             callbackFn(_this, _arguments);
             last = now;
         } else if (timer === null && trailing) {
             timer = setTimeout(function () {
                 timer = null;
                 callbackFn(_this, _arguments);
             }, duration);
         }
     }

     handleFn.cancel = function(){
         if (timer) {
             clearTimeout(timer);
             timer = null;   //先执行，再清除
         }
     }

     function callbackFn(context, arguments) {
         var res = fn.apply(context, arguments);
         if (resultFn && resultFn instanceof Function) {
             resultFn(res);
         }
     }

     return handleFn;
 }
